import yfinance as yf from supabase import create_client, Client import pandas as pd import time import os import pytz from datetime import datetime from dotenv import load_dotenv import yahoo_fin.stock_info as si # Load environment variables from .env file (if present) load_dotenv() # --------------------------------------------------------- # CONSTANTS & CONFIG # --------------------------------------------------------- SUPABASE_URL = os.environ.get("SUPABASE_URL") SUPABASE_KEY = os.environ.get("SUPABASE_SERVICE_ROLE_KEY") BATCH_SIZE = 50 MAX_HISTORY_POINTS = 365 * 5 # Cap: 5 years of daily points if not SUPABASE_URL or not SUPABASE_KEY: print("ERROR: SUPABASE_URL and SUPABASE_SERVICE_ROLE_KEY must be set in .env") exit(1) supabase: Client = create_client(SUPABASE_URL, SUPABASE_KEY) print("Successfully connected to Supabase.") # Global state tracker for end-of-day history dump history_written_today = False current_date_str = "" def get_nyse_tickers(): """Fetches all active US tickers using yahoo_fin with validation.""" print("Fetching active US tickers...") try: other = si.tickers_other() nasdaq = si.tickers_nasdaq() tickers = list(set(other + nasdaq)) # Clean and validate tickers more aggressively clean_tickers = [] invalid_count = 0 for t in tickers: if not isinstance(t, str) or len(t) == 0: invalid_count += 1 continue # Clean the ticker: strip whitespace, replace . with -, remove $ and other invalid chars clean = t.strip().replace('.', '-').replace('$', '') # Validate: only alphanumeric, hyphens, and underscores allowed # Most US tickers are 1-5 characters, but some can be longer if not clean or len(clean) > 10: invalid_count += 1 continue # Skip if it still contains invalid characters if not all(c.isalnum() or c == '-' for c in clean): invalid_count += 1 continue clean_tickers.append(clean) print(f"Found {len(clean_tickers)} valid US tickers (discarded {invalid_count} invalid).") return clean_tickers except Exception as e: print(f"Error fetching tickers: {e}") print("Using fallback ticker list...") return ["AAPL", "MSFT", "GOOGL", "AMZN", "META"] def safe_float(value, fallback=0.0): """Safely cast a value to float, handling pandas NaN.""" try: val = float(value) if pd.isna(val): return fallback return round(val, 2) except (TypeError, ValueError): return fallback def is_market_open_cst(): """Returns True if it's currently between 8:30 AM and 3:00 PM CST M-F.""" cst = pytz.timezone('US/Central') now = datetime.now(cst) # Weekends are closed if now.weekday() >= 5: return False market_open_time = datetime.strptime("08:30", "%H:%M").time() market_close_time = datetime.strptime("15:00", "%H:%M").time() return market_open_time <= now.time() <= market_close_time def market_closed_for_day_cst(): """Returns True if the current time is strictly after 3:00 PM CST (End of Day).""" cst = pytz.timezone('US/Central') now = datetime.now(cst) market_close_time = datetime.strptime("15:00", "%H:%M").time() return now.time() > market_close_time def write_history_for_all(tickers_list): """ Called exactly once per day after market close. Iterates through all stocks, fetches the latest price, and appends it to stock_history. """ print("--- INITIATING END OF DAY HISTORY FLUSH ---") today_str = datetime.now(pytz.timezone('US/Central')).strftime('%Y-%m-%d') for i in range(0, len(tickers_list), BATCH_SIZE): batch = tickers_list[i:i+BATCH_SIZE] try: # 1. Fetch current live prices from `stocks` table stocks_res = supabase.table("stocks").select("symbol, price").in_("symbol", batch).execute() current_prices = {row["symbol"]: row.get("price", 0.0) for row in stocks_res.data} # 2. Fetch existing history from `stock_history` history_res = supabase.table("stock_history").select("symbol, prices").in_("symbol", batch).execute() history_dict = {row["symbol"]: row.get("prices", []) for row in history_res.data} upserts = [] for sym in batch: if sym not in current_prices: continue cp = current_prices[sym] hp_list = history_dict.get(sym, []) # Double-check if we accidentally already recorded today if hp_list and hp_list[-1].get("date") == today_str: hp_list[-1]["price"] = cp else: hp_list.append({"date": today_str, "price": cp}) # Trim to 5 years max points hp_list = hp_list[-MAX_HISTORY_POINTS:] upserts.append({ "symbol": sym, "prices": hp_list, "updatedAt": "now()" }) if upserts: supabase.table("stock_history").upsert(upserts).execute() print(f" Flushed history for batch {i//BATCH_SIZE + 1}...") except Exception as e: print(f"Error writing history batch: {e}") time.sleep(1) # Prevent rate limits print("--- END OF DAY HISTORY FLUSH COMPLETE ---") def main_loop(): global history_written_today, current_date_str tickers = get_nyse_tickers() print("Beginning continuous update loop...") while True: cst = pytz.timezone('US/Central') now_cst = datetime.now(cst) today_date = now_cst.strftime('%Y-%m-%d') # Reset the daily history boolean flag on a new calendar day if today_date != current_date_str: current_date_str = today_date history_written_today = False market_open = is_market_open_cst() sleep_time = 5 if market_open else 30 print(f"\n[CYCLE START] Stocks: {len(tickers)} | Market Open (CST): {market_open} | Batch Delay: {sleep_time}s | Time: {now_cst.strftime('%H:%M:%S')}") for i in range(0, len(tickers), BATCH_SIZE): batch = tickers[i:i + BATCH_SIZE] batch_str = " ".join(batch) try: data = yf.Tickers(batch_str) upserts = [] for ticker in batch: try: t = data.tickers[ticker] # Fetch stock info info = t.info # Skip if info is empty or None if not info: continue # Grab the price safely cp = 0.0 if 'currentPrice' in info: cp = safe_float(info['currentPrice']) elif 'regularMarketPrice' in info: cp = safe_float(info['regularMarketPrice']) # Fallback to daily history if 'info' lacks immediate price if cp == 0.0: try: hist = t.history(period="1d") if not hist.empty: cp = safe_float(hist['Close'].iloc[-1]) else: continue # Unable to get any price, skip this ticker except Exception: continue # Skip if history fetch fails prev_close = safe_float(info.get('previousClose') or cp) change = round(cp - prev_close, 2) change_pct = round((change / prev_close) * 100, 4) if prev_close else 0.0 vol = info.get('volume') if vol is None or pd.isna(vol): vol = 0 # Only upsert if we have a valid price if cp > 0: upserts.append({ "symbol": ticker, "name": info.get('shortName', ticker), "price": cp, "change": change, "changePercent": change_pct, "volume": int(vol), "dayHigh": safe_float(info.get('dayHigh')), "dayLow": safe_float(info.get('dayLow')), "high52w": safe_float(info.get('fiftyTwoWeekHigh')), "low52w": safe_float(info.get('fiftyTwoWeekLow')), "bid": safe_float(info.get('bid')), "ask": safe_float(info.get('ask')), "updatedAt": "now()" }) except KeyError: # Ticker not found in response pass except Exception as e: # Skip individual tickers that fail pass if upserts: supabase.table("stocks").upsert(upserts).execute() print(f" Upserted batch {i//BATCH_SIZE + 1} ({len(upserts)} records) - Sleeping {sleep_time}s") else: print(f" Batch {i//BATCH_SIZE + 1} had no valid data - Sleeping {sleep_time}s") except Exception as e: print(f" Batch Request Error: {e}") time.sleep(sleep_time) # Recalculate market state mid-loop in case it flips during this cycle now_cst = datetime.now(cst) market_open = is_market_open_cst() sleep_time = 5 if market_open else 30 # After finishing a FULL PASS over all stocks... # Check if the market HAS CLOSED for the day and we need to lock in the history point if market_closed_for_day_cst() and not history_written_today: write_history_for_all(tickers) history_written_today = True if __name__ == "__main__": main_loop()
